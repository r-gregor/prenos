# Sort keys before iterating
echo -e "\nSorted Inventory:"
echo "----------------"
IFS=$'\n' sorted_keys=($(sort <<< "${!inventory[*]}"))
for fruit in "${sorted_keys[@]}"; do
	printf "%-10s: %d\n" "$fruit" "${inventory[$fruit]}"
done
